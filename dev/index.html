Quickstart

using ChainLadder
using CSV
using Test
using DataFrames

raa = CSV.File("test/data/raa.csv") |> DataFrame

t = CumulativeTriangle(raa.origin,raa.development,raa.values)

lin = LossDevelopmentFactor(t)

s = square(t,lin)

total_loss(t,lin)

outstanding_loss(t,lin)
